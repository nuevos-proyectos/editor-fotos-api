from flask import Flask, request, send_file
from PIL import Image, ImageDraw, ImageFont
import io

app = Flask(__name__)

@app.route('/editar', methods=['POST'])
def editar_foto():
    # Recibir la imagen y el texto desde n8n
    if 'foto' not in request.files:
        return "Falta la foto", 400
    
    file = request.files['foto']
    texto_datos = request.form.get('texto', 'Sin datos')

    # Abrir la imagen original
    img = Image.open(file.stream).convert("RGBA")
    ancho, alto = img.size

    # Crear una capa transparente para el recuadro
    capa_recuadro = Image.new('RGBA', img.size, (255,255,255,0))
    dibujo = ImageDraw.Draw(capa_recuadro)

    # Coordenadas y tamaño del recuadro (Ajustable)
    margen = 20
    ancho_recuadro = 400
    alto_recuadro = 150
    x1 = margen
    y1 = alto - alto_recuadro - margen
    x2 = x1 + ancho_recuadro
    y2 = alto - margen

    # Dibujar el rectángulo blanco al 70% de opacidad (180 sobre 255)
    dibujo.rectangle(((x1, y1), (x2, y2)), fill=(255, 255, 255, 180))

    # Escribir el texto encima del recuadro
    # (Pillow usa una fuente por defecto si no le pasas un archivo .ttf)
    dibujo.text((x1 + 15, y1 + 15), texto_datos, fill=(0, 0, 0, 255))

    # Unir la foto original con la capa del recuadro
    imagen_final = Image.alpha_composite(img, capa_recuadro)
    imagen_final = imagen_final.convert("RGB") # Convertir de nuevo a formato JPG estándar

    # Guardar en memoria y devolver a n8n
    img_io = io.BytesIO()
    imagen_final.save(img_io, 'JPEG', quality=85)
    img_io.seek(0)

    return send_file(img_io, mimetype='image/jpeg')

if __name__ == '__main__':
    # Railway inyecta su propio puerto automáticamente
    import os
    port = int(os.environ.get("PORT", 5000))
    app.run(host='0.0.0.0', port=port)
